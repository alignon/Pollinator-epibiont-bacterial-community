## Phylogenetic Tree Construction
sequences <- getSequences('ASVs.fa')
alignment <- AlignSeqs(DNAStringSet(sequences), anchor=NA)

# Convert alignment into phyDat format

# Compute maximum-likelihood distance matrix and build NJ tree
dm <- dist.ml(phang.align) 
treeNJ <- NJ(dm) 

# Fit a basic phylogenetic model using GTR
fit <- pml(treeNJ, data = phang.align) 
fitGTR <- update(fit, k=4, inv=0.2)

# Optimise GTR tree fit

                    rearrangement = "stochastic", control = pml.control(trace = 0))

## DESEQ2 - Cycles through all pairwise treatment combinations for the chosen year
## NOTE: ps.2 must be your raw counts phyloseq object (before CSS normalisation)

deseq_results_list <- list()  # Store all results

for (combo in alpha_comparisons) {
  
  grp1 <- combo[1]
  grp2 <- combo[2]
  label <- paste0(gsub(" ", "_", grp1), "_vs_", gsub(" ", "_", grp2))
  message("\n========================================")
  message("Running DESeq2: ", grp1, " vs ", grp2)
  message("========================================")
  
  # Subset to the two treatments
  ps.sub <- subset_samples(ps.2, Treatment %in% c(grp1, grp2))
  ps.sub <- prune_taxa(taxa_sums(ps.sub) > 0, ps.sub)
  
  # Make Treatment a factor with grp2 as reference (positive LFC = grp1 enriched)
  sample_data(ps.sub)$Treatment <- factor(
    sample_data(ps.sub)$Treatment,
    levels = c(grp2, grp1)
  )
  
  # Run DESeq2
  dds <- phyloseq_to_deseq2(ps.sub, ~ Treatment)
  dds <- DESeq(dds, sfType = "poscounts")
  
  # Extract results
  res <- results(dds)
  res <- res[order(res$padj, na.last = NA), ]
  
  # Print full results for this comparison
  message("\n--- Full DESeq2 results (", grp1, " vs ", grp2, ") ---")
  print(res)
  
  alpha <- 0.01
  sigtab <- res[!is.na(res$padj) & res$padj < alpha, ]
  
  if (nrow(sigtab) == 0) {
    message("  No significant taxa (padj < 0.01) for: ", label)
    next
  }
  
  sigtab <- cbind(
    as(sigtab, "data.frame"),
    as(tax_table(ps.sub)[rownames(sigtab), ], "matrix")
  )
  
  sigtab$ASV <- rownames(sigtab)
  
  # Print significant results summary for this comparison
  message("\n--- Significant taxa (padj < 0.01): ", nrow(sigtab), " ---")
  print(sigtab)
  
  deseq_results_list[[label]] <- sigtab  # Save for later use
  
  # --- Plotting ---
  sigtabgen <- subset(sigtab, !is.na(Family))
  
  if (nrow(sigtabgen) == 0) {
    message("  No taxa with Family annotation to plot for: ", label)
    next
  }
  
  # Order by Phylum then Genus
  x <- tapply(sigtabgen$log2FoldChange, sigtabgen$Phylum, max)
  sigtabgen$Phylum <- factor(as.character(sigtabgen$Phylum), levels = names(sort(x, TRUE)))
  
  x <- tapply(sigtabgen$log2FoldChange, sigtabgen$Genus, max)
  sigtabgen$Genus <- factor(as.character(sigtabgen$Genus), levels = names(sort(x, TRUE)))
  
  max_lfc <- max(abs(sigtabgen$log2FoldChange), na.rm = TRUE)
  
  # Direction label: positive LFC = grp1, negative LFC = grp2
  sigtabgen$Direction <- ifelse(sigtabgen$log2FoldChange > 0, grp1, grp2)
  
  # Colour: only the two relevant treatments from the year palette
  combo_colour <- pollinator_colour[c(grp2, grp1)]
  
  # Plot: Genus + ASV ID
  sigtabgen$Genus_ASV <- paste0(sigtabgen$Genus, "\n", sigtabgen$ASV)
  sigtabgen$Genus_ASV <- factor(
    sigtabgen$Genus_ASV,
    levels = sigtabgen$Genus_ASV[order(sigtabgen$log2FoldChange)]
  )
  
  p2 <- ggplot(sigtabgen, aes(y = Genus_ASV, x = log2FoldChange, fill = Direction)) +
    geom_vline(xintercept = 0, color = "darkgray", linewidth = 0.5) +
    geom_col() +
    scale_fill_manual(values = combo_colour) +
    scale_x_continuous(limits = c(-max_lfc, max_lfc)) +
    theme_bw() +
    theme(axis.text.y = element_text(face = "italic", lineheight = 0.9)) +
    labs(
      title = paste0("DESeq2: ", grp1, " vs ", grp2, " (", YEAR, ")"),
      x = paste0("\u2190 ", grp2, " | log2FC | ", grp1, " \u2192"),
      y = "Genus (ASV ID)",
      fill = ""
    )
  print(p2)
  
  message("\n--- Completed: ", grp1, " vs ", grp2, " ---\n")
  
}

# Summary: list all comparisons with significant results
message("\n========================================")
message("DESeq2 complete. Comparisons with significant taxa:")
print(names(deseq_results_list))
message("========================================")


## GENUS OTU FREQUENCY TABLE
## Identifies how frequently each genus appears across samples (presence/absence)
## and flags whether each genus is gut-associated or environmental

# Extract OTU table (samples x OTUs)
otu <- as(otu_table(ps_rel_abund.pol), "matrix")
if (taxa_are_rows(ps_rel_abund.pol)) otu <- t(otu)

# Presence / absence
otu_pa <- otu > 0

# Taxonomy
tax <- as(tax_table(ps_rel_abund.pol), "matrix")

# Extract sample metadata
meta <- data.frame(sample_data(ps_rel_abund.pol))
if (!"Sample" %in% colnames(meta)) {
  meta$Sample <- rownames(meta)
}

# Long format + add Genus + Treatment
df <- as.data.frame(otu_pa) |>
  rownames_to_column("Sample") |>
  pivot_longer(-Sample, names_to = "OTU", values_to = "Present") |>
  left_join(
    data.frame(OTU = rownames(tax), Genus = tax[, "Genus"]),
    by = "OTU"
  ) |>
  left_join(
    meta[, c("Sample", "Treatment")],
    by = "Sample"
  )

# Frequency by genus
genus_frequency <- df |>
  filter(Present, !is.na(Genus)) |>
  group_by(Genus) |>
  summarise(
    n_samples = n(),
    Treatments_found_in = paste(sort(unique(Treatment)), collapse = ", "),
    n_treatments = n_distinct(Treatment),
    .groups = "drop"
  )
print(genus_frequency)

# Define gut-associated genera of interest
specific_genera.pol <- c(
  "Acinetobacter", "Apibacter", "Apilactobacillus", "Arsenophonus",
  "Bartonella", "Bifidobacterium", "Bombella", "Bombilactobacillus",
  "Bombiscardovia", "Commensalibacter", "Frischella", "Gilliamella",
  "Lactobacillus", "Lysinibacillus", "Melissococcus", "Paenibacillus",
  "Parasaccharibacter", "Schmidhempelia", "Serratia", "Snodgrassella",
  "Spiroplasma", "Caulobacter", "Escherichia", "Pseudomonas",
  "Staphylococcus"
)

# Top 51 genera by sample frequency
top51_genus <- genus_frequency |>
  filter(!is.na(Genus)) |>
  arrange(desc(n_samples)) |>
  slice_head(n = 51)

# Flag gut-associated vs environmental
top51_genus <- top51_genus |>
  mutate(
    Association = if_else(
      Genus %in% specific_genera.pol,
      "Gut-associated",
      "Environmental"
    )
  )

# Rename for export
worksheet_top51 <- top51_genus |>
  rename(OTU_sample_frequency = n_samples)

print(worksheet_top51)

# Export to CSV (filename uses YEAR variable so it updates automatically)
write.csv(
  worksheet_top51,
  file = paste0("Top51_Genus_OTU_Frequency_Gut_vs_Environmental_", YEAR, ".csv"),
  row.names = FALSE
)
message("Exported: Top51_Genus_OTU_Frequency_Gut_vs_Environmental_", YEAR, ".csv")
